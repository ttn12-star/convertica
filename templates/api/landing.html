import requests
r = requests.post(
    "https://convertica.net/api/v1/pdf-to-word/",
    headers={"Authorization": "Bearer cvk_live_xxxxxxxxxxxxxxxx"},
    files={"pdf_file": open("input.pdf", "rb")},
)
with open("output.docx", "wb") as f:
    f.write(r.content)
